int add(int a,int b=10, int c =20){
    return a+b+c;
}

float power(float base,int exp=2){
    float res=1.0;
    for(int i=0;i<exp;i++){
        res*=base;
    }
    return res;
}

bool is_even(int n){
    return (n%2)==0;
}

bool toggle(bool v=true){
    return !v;
}
